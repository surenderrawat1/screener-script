            if (netEl) netEl.textContent = '₹' + Math.round(pf.net_pnl || 0).toLocaleString('en-IN');
            if (netWrap) netWrap.className = 'Net ' + pnlClass(pf.net_pnl);
        } else if (pfBox) {
            pfBox.hidden = true;
        }
    }

    function updateAlert(pos) {
        const alert = el('swp-pos-alert');
        if (!alert) return;
        const urgent = (pos?.rows || []).filter(r => r.exit_verdict === 'EXIT');
        if (urgent.length) {
            alert.hidden = false;
            alert.textContent = urgent.length + ' position(s) triggered EXIT — '
                + urgent.map(r => r.symbol + ' (' + (r.exit_triggers || []).join(', ') + ')').join('; ');
        } else {
            alert.hidden = true;
        }
    }

    function updateRows(pos) {
        const body = el('swp-open-body');
        const empty = el('swp-empty-msg');
        const wrap = el('swp-open-wrap');
        if (!body) return;

        const rows = pos?.rows || [];
        updateSummary(pos);
        updateAlert(pos);

        if (!rows.length) {
            if (empty) empty.hidden = false;
            if (wrap) wrap.hidden = true;
            return;
        }
        if (empty) empty.hidden = true;
        if (wrap) wrap.hidden = false;

        rows.forEach(r => {
            const tr = body.querySelector('tr[data-pos-id="' + r.id + '"]');
            if (!tr) return;

            const nowEl = tr.querySelector('.swp-now');
            if (nowEl) {
                const priceKey = r.current_price != null ? String(r.current_price) : '';
                if (priceKey && nowEl.dataset.price !== priceKey) {
                    nowEl.classList.add('swp-flash');
                    setTimeout(() => nowEl.classList.remove('swp-flash'), 650);
                }
                nowEl.dataset.price = priceKey;
                nowEl.innerHTML = renderNow(r);
            }

            const pnlEl = tr.querySelector('.swp-pnl');
            if (pnlEl) {
                const pctEl = pnlEl.querySelector('.swp-pnl-pct');
                const grossEl = pnlEl.querySelector('.swp-pnl-gross');
                if (r.gain_pct != null && pctEl) {
                    const cls = pnlClass(r.gain_pct);
                    pctEl.className = 'swp-pnl-pct ' + cls;
                    pctEl.textContent = r.gain_pct + '%';
                    if (grossEl) {
                        grossEl.className = 'swp-pnl-gross ' + cls;
                        grossEl.style.fontSize = '0.65rem';
                        grossEl.textContent = r.pnl_inr != null ? ' · gross ₹' + Math.round(r.pnl_inr).toLocaleString('en-IN') : '';
                    }
                }
            }

            const sessEl = tr.querySelector('.swp-sessions');
            if (sessEl) sessEl.textContent = r.sessions_held != null ? String(r.sessions_held) : '—';

            const exitEl = tr.querySelector('.swp-exit-cell');
            if (exitEl) {
                const v = r.exit_verdict || 'HOLD';
                exitEl.innerHTML = '<span class="swp-verdict ' + verdictClass(v) + '">' + esc(v) + '</span>';
            }

            const trigEl = tr.querySelector('.swp-triggers-cell');
            if (trigEl) {
                const t = r.exit_triggers || [];
                trigEl.textContent = t.length ? t.join(', ') : '—';
            }

            const lvlEl = tr.querySelector('.swp-levels');
            if (lvlEl) lvlEl.innerHTML = renderLevels(r);

            const closeInput = tr.querySelector('input[name="closed_price"]');
            if (closeInput && !closeInput.value && r.current_price != null) {
                closeInput.value = String(r.current_price);
            }
        });
    }

    function setStatus(ok) {
        const pill = el('swp-status-pill');
        if (!pill) return;
        pill.classList.remove('swp-live', 'swp-error');
        if (ok) {
            pill.textContent = 'Live · auto';
            pill.classList.add('swp-live');
        } else {
            pill.textContent = 'Connection error';
            pill.classList.add('swp-error');
        }
    }

    function tickCountdown() {
        const cd = el('swp-countdown');
        if (!cd) return;
        const sec = Math.max(0, Math.ceil((nextRefreshAt - Date.now()) / 1000));
        cd.textContent = sec + 's';
    }

    async function poll(refresh) {
        if (polling) return;
        polling = true;
        const btn = el('swp-refresh-btn');
        if (btn && refresh) btn.disabled = true;
        try {
            const r = await fetch(API + (refresh ? '?refresh=1' : ''), { credentials: 'same-origin' });
            if (!r.ok) throw new Error('HTTP ' + r.status);
            const data = await r.json();
            if (!data.ok) throw new Error(data.error || 'API error');
            if (data.positions) updateRows(data.positions);
            const updated = el('swp-updated');
            if (updated && data.positions?.refreshed_at) {
                updated.textContent = new Date(data.positions.refreshed_at).toLocaleTimeString();
            }
            setStatus(true);
            nextRefreshAt = Date.now() + REFRESH_MS;
        } catch (e) {
            setStatus(false);
        } finally {
            polling = false;
            if (btn) btn.disabled = false;
        }
    }

    const refreshBtn = el('swp-refresh-btn');
    if (refreshBtn) refreshBtn.addEventListener('click', () => poll(true));

    setInterval(tickCountdown, 1000);
    setInterval(() => poll(false), REFRESH_MS);
    tickCountdown();
    if (el('swp-open-body')) poll(true);
})();

(function () {
    function symbolFromBtn(btn) {
        const form = btn.closest('form');
        if (!form) return '';
        const named = btn.getAttribute('data-symbol-input');
        const symInput = named
            ? (form.querySelector('[name="' + named + '"]') || document.querySelector('[name="' + named + '"]'))
            : form.querySelector('[name="symbol"]');
        return symInput ? String(symInput.value || '').trim().toUpperCase() : '';
    }

    document.querySelectorAll('.swp-fetch-price').forEach(function (btn) {
        btn.addEventListener('click', function () {
            const form = btn.closest('form');
            const priceInput = form ? form.querySelector('[name="entry_price"]') : null;
            const symbol = symbolFromBtn(btn);
            if (!priceInput || symbol === '') {
                alert('Enter a symbol first.');
                return;
            }
            const label = btn.textContent;
            btn.disabled = true;
            btn.textContent = '…';
            fetch('swing-live-price.php?symbol=' + encodeURIComponent(symbol))
                .then(function (r) { return r.json(); })
                .then(function (data) {
                    if (!data.success || !data.price) {
                        throw new Error(data.error || 'No quote');
                    }
                    priceInput.value = Number(data.price).toFixed(2);
                })
                .catch(function (e) {
                    alert(e.message || 'Live fetch failed');
                })
                .finally(function () {
                    btn.disabled = false;
                    btn.textContent = label;
                });
        });
    });
})();
