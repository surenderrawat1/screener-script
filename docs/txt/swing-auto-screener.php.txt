                if (r.stop_distance_pct != null) parts.push('<span class="sas-trail-bar">' + r.stop_distance_pct + '% above floor</span>');
                if (r.profit_target != null) parts.push('Tgt ₹' + Number(r.profit_target).toLocaleString('en-IN', {minimumFractionDigits:2}));
                if (r.breakeven_armed) parts.push('<span class="sas-trail-armed">BE+</span>');
                return parts.join('<br>') || '—';
            })();
            const triggers = (r.exit_triggers || []).join(', ');
            const now = r.ok
                ? fmtRs(r.current_price) + (window.PriceFreshness ? window.PriceFreshness.renderBadge(r) : '')
                : '<span class="sas-neg">' + esc(r.error || 'No data') + '</span>';
            const pnl = r.gain_pct != null
                ? '<span class="' + gainCls + '">' + fmtPct(r.gain_pct) + '</span>' + (r.net_pnl != null ? '<br><span class="' + gainCls + '" style="font-size:0.65rem;">Net ₹' + Math.round(r.net_pnl).toLocaleString('en-IN') + '</span>' : '')
                : '—';
            const act = r.position_action || 'REVIEW';
            const reasons = (r.action_reasons || []).join(' · ');
            return '<tr>'
                + '<td>' + symLink + (r.notes ? '<div class="sas-muted">' + esc(r.notes) + '</div>' : '') + '</td>'
                + '<td>' + fmtRs(r.entry_price) + '<br><span class="sas-muted">' + esc(r.entry_date) + '</span></td>'
                + '<td>' + now + '</td>'
                + '<td>' + pnl + '</td>'
                + '<td>' + (r.r_unrealized != null ? r.r_unrealized + 'R' : '—') + '</td>'
                + '<td><span class="sas-decision ' + decisionClass(act) + '">' + esc(r.action_label || act) + '</span>'
                + (reasons ? '<div class="sas-muted" style="font-size:0.62rem;margin-top:0.2rem;">' + esc(reasons) + '</div>' : '') + '</td>'
                + '<td><span class="sas-verdict ' + verdictClass(r.exit_verdict) + '">' + esc(r.exit_verdict) + '</span>'
                + (triggers ? '<div class="sas-muted" style="font-size:0.62rem;">' + esc(triggers) + '</div>' : '') + '</td>'
                + '<td style="font-size:0.68rem;">' + stopTgt + '</td>'
                + '</tr>';
        }).join('');
    }

    function btClass(grade) {
        const g = String(grade || '').toLowerCase();
        if (g === 'strong') return 'sas-bt-strong';
        if (g === 'ok') return 'sas-bt-ok';
        if (g === 'weak') return 'sas-bt-weak';
        if (g === 'fail') return 'sas-bt-fail';
        return 'sas-bt-unproven';
    }

    function renderBacktestCell(h) {
        if (!h.backtest_grade) {
            return '<span class="sas-muted">—</span>';
        }
        const pf = h.backtest_pf != null ? 'PF ' + h.backtest_pf : '';
        const wr = h.backtest_win_rate_pct != null ? h.backtest_win_rate_pct + '%' : '';
        const tr = h.backtest_trades ? h.backtest_trades + 't' : '';
        const sub = [pf, wr, tr].filter(Boolean).join(' · ');
        return '<span class="sas-bt ' + btClass(h.backtest_grade) + '">' + esc(h.backtest_label || h.backtest_grade) + '</span>'
            + (sub ? '<span class="sas-muted" style="font-size:0.58rem;display:block;">' + esc(sub) + '</span>' : '');
    }

    function renderHits() {
        const body = el('sas-hits-body');
        if (!body) return;
        const hits = tierData[activeTier] || [];
        [
            ['hi', 'high_conviction'], ['strict', 'strict_enter'], ['setup', 'setup_radar'], ['breakout', 'breakout_surge']
        ].forEach(([id, key]) => {
            const n = el('sas-n-' + id);
            if (n) n.textContent = String((tierData[key] || []).length);
        });
        if (!hits.length) {
            const labels = {
                high_conviction: 'high conviction',
                strict_enter: 'Strict ENTER',
                setup_radar: 'SETUP+',
                breakout_surge: 'breakout + volume surge'
            };
            body.innerHTML = '<tr><td colspan="15" class="sas-muted">No ' + esc(labels[activeTier] || 'hits') + ' in latest scan.</td></tr>';
            return;
        }
        body.innerHTML = hits.map(h => {
            const zone = h.ta_52w_chart_zone || '';
            const zoneCls = zone === 'green' ? 'sas-zone-green' : (zone === 'red' ? 'sas-zone-red' : '');
            const pct = h.ta_pct_52w != null ? h.ta_pct_52w + '%' + (zone ? ' · ' + zone.toUpperCase() : '') : '—';
            const sym = '<a class="sas-link" href="swing-trading.php?run=1&amp;mode=symbol&amp;symbol=' + encodeURIComponent(h.symbol) + '">' + esc(h.symbol) + '</a>'
                + (h.already_held ? '<span class="sas-held' + (h.held_near_stop ? ' sas-held-near' : '') + '">Held'
                    + (h.held_action_label ? ' · ' + esc(h.held_action_label) : '')
                    + (h.held_stop_distance_pct != null ? ' · ' + h.held_stop_distance_pct + '% above stop' : '')
                    + '</span>' : '');
            const flags = (h.risk_flags || []).map(f => '<span class="sas-flag">' + esc(f.replace(/_/g, ' ')) + '</span>').join('') || '—';
            const dAct = h.decision_action || 'SKIP';
            const shares = h.suggested_shares ?? 0;
            const showAdd = h.add_allowed && canAddPortfolio && shares > 0 && !h.already_held;
            const busy = addingSymbol === h.symbol;
            const addCell = showAdd
                ? '<button type="button" class="sas-add-btn" data-add="' + esc(h.symbol) + '"' + (busy ? ' disabled' : '') + '>+ Add</button>'
                    + '<span class="sas-add-shares">' + shares + ' sh · ~₹' + Math.round(shares * (h.price || 0)).toLocaleString('en-IN') + '</span>'
                : '<span class="sas-muted">—</span>';
            return '<tr>'
                + '<td>' + sym + '</td>'
                + '<td><span class="sas-decision ' + decisionClass(dAct) + '">' + esc(h.decision_label || dAct) + '</span></td>'
                + '<td><strong>' + (h.decision_score ?? 0) + '</strong></td>'
                + '<td>' + renderBacktestCell(h) + '</td>'
                + '<td>' + (h.swing_rank ?? 0) + '</td>'
                + '<td>' + (h.entry_score ?? 0) + '</td>'
                + '<td><span class="sas-verdict ' + verdictClass(h.strict) + '">' + esc(h.strict) + '</span></td>'
                + '<td>' + (h.r_multiple ?? '—') + '</td>'
                + '<td>' + fmtRs(h.price) + '</td>'
                + '<td>' + fmtRs(h.stop_loss) + '</td>'
                + '<td>' + fmtRs(h.profit_target) + '</td>'
                + '<td>' + (h.ta_rsi14 ?? '—') + '</td>'
                + '<td class="' + zoneCls + '">' + esc(pct) + '</td>'
                + '<td>' + flags + '</td>'
                + '<td>' + addCell + '</td>'
                + '</tr>';
        }).join('');
        body.querySelectorAll('[data-add]').forEach(btn => {
            btn.addEventListener('click', () => {
                const sym = btn.getAttribute('data-add');
                const hit = hits.find(x => x.symbol === sym);
                if (hit) addFromHit(hit);
            });
        });
    }

    async function postJson(action, payload) {
        const r = await fetch(API + '?action=' + encodeURIComponent(action), {
            method: 'POST',
            credentials: 'same-origin',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ ...payload, _csrf: CSRF }),
        });
        const data = await r.json().catch(() => ({}));
        if (!r.ok && !data.error) throw new Error('HTTP ' + r.status);
        return data;
    }

    async function addFromHit(hit) {
        if (!hit || addingSymbol) return;
        const shares = hit.suggested_shares ?? 0;
        const msg = 'Add ' + hit.symbol + '?\n'
            + 'Entry ₹' + Number(hit.price).toFixed(2)
            + ' · Stop ₹' + Number(hit.stop_loss).toFixed(2)
            + ' · ' + shares + ' shares'
            + (hit.profit_target ? ' · Target ₹' + Number(hit.profit_target).toFixed(2) : '');
        if (!window.confirm(msg)) return;

        addingSymbol = hit.symbol;
        renderHits();
        try {
            const check = await postJson('check_add', {
                symbol: hit.symbol,
                entry_price: hit.price,
                stop_loss: hit.stop_loss,
                shares: shares,
            });
            if (!check.ok) {
                window.alert(check.error || 'Cannot add — heat or regime gate blocked.');
                return;
            }
            const added = await postJson('add_position', {
                symbol: hit.symbol,
                entry_price: hit.price,
                stop_loss: hit.stop_loss,
                profit_target: hit.profit_target || null,
                shares: check.shares ?? shares,
                as_of_date: hit.as_of_date || null,
                decision_action: hit.decision_action,
                decision_label: hit.decision_label,
                decision_score: hit.decision_score,
            });
            if (!added.ok) {
                window.alert(added.error || 'Add failed.');
                return;
            }
            if (added.positions) renderPositions(added.positions);
            const pill = el('sas-status-pill');
            if (pill) {
                pill.textContent = added.message || ('Added ' + hit.symbol);
                pill.classList.add('sas-live');
            }
        } catch (e) {
            window.alert('Add failed: ' + e.message);
        } finally {
            addingSymbol = '';
            renderHits();
        }
    }

    function updateMeta(data) {
        setStatus(data.scan);
        renderGuidance(data);
        renderKpi(data);
        if (data.tiers) {
            tierData = data.tiers;
            renderHits();
        }
        if (data.positions) renderPositions(data.positions);
        const updated = el('sas-updated');
        if (updated) {
            const t = data.positions?.refreshed_at || data.scan?.saved_at;
            updated.textContent = t ? new Date(t).toLocaleTimeString() : '—';
        }
        nextScanAt = Date.now() + (data.next_scan_in_sec ?? SCAN_MS / 1000) * 1000;
        nextFullScanAt = Date.now() + (data.next_full_scan_in_sec ?? FULL_SCAN_MS / 1000) * 1000;
    }

    function tickCountdown() {
        const cd = el('sas-countdown');
        if (!cd || !nextScanAt) return;
        const sec = Math.max(0, Math.ceil((nextScanAt - Date.now()) / 1000));
        cd.textContent = sec + 's';
    }

    function tickFullCountdown() {
        const cd = el('sas-full-countdown');
        if (!cd || !nextFullScanAt) return;
        const sec = Math.max(0, Math.ceil((nextFullScanAt - Date.now()) / 1000));
        const m = Math.floor(sec / 60);
        const s = sec % 60;
        cd.textContent = m > 0 ? m + 'm ' + s + 's' : sec + 's';
    }

    async function fetchJson(url) {
        const r = await fetch(url, { credentials: 'same-origin' });
        if (!r.ok) throw new Error('HTTP ' + r.status);
        return r.json();
    }

    async function loadState(opts = {}) {
        const positions = opts.positions !== false ? '1' : '0';
        return fetchJson(API + '?action=state&positions=' + positions);
    }

    async function refreshPositionsOnly() {
        try {
            const data = await fetchJson(API + '?action=positions');
            if (data.positions) {
                renderPositions(data.positions);
                const updated = el('sas-updated');
                if (updated && data.positions.refreshed_at) {
                    updated.textContent = new Date(data.positions.refreshed_at).toLocaleTimeString();
                }
            }
        } catch (e) { /* silent — full cycle will retry */ }
    }

    async function startScan() {
        return postJson('start', {});
    }

    function isScanning(data) {
        const st = data?.scan?.status;
        return st === 'pending' || st === 'running';
    }

    function schedulePoll() {
        if (pollTimer) clearTimeout(pollTimer);
        pollTimer = setTimeout(async () => {
            try {
                const data = await loadState({ positions: false });
                updateMeta(data);
                if (isScanning(data)) {
                    schedulePoll();
                } else {
                    const full = await loadState({ positions: true });
                    updateMeta(full);
                }
            } catch (e) {
                const pill = el('sas-status-pill');
                if (pill) { pill.textContent = 'Connection error'; pill.classList.add('sas-scanning'); }
                schedulePoll();
            }
        }, POLL_MS);
    }

    async function scanCycle() {
        try {
            let data = await loadState({ positions: false });
            updateMeta(data);
            if (data.should_start_scan) {
                data = await startScan();
                updateMeta(data);
            }
            if (isScanning(data)) {
                schedulePoll();
            } else {
                const full = await loadState({ positions: true });
                updateMeta(full);
            }
        } catch (e) {
            const pill = el('sas-status-pill');
            if (pill) pill.textContent = 'Error: ' + e.message;
        }
    }

    document.querySelectorAll('#sas-tier-tabs button').forEach(btn => {
        btn.addEventListener('click', () => {
            document.querySelectorAll('#sas-tier-tabs button').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
            activeTier = btn.dataset.tier || 'strict_enter';
            renderHits();
        });
    });

    scanCycle();
    const urlTier = new URLSearchParams(window.location.search).get('tier');
    const validTiers = ['high_conviction', 'strict_enter', 'setup_radar', 'breakout_surge'];
    if (urlTier && validTiers.includes(urlTier)) {
        activeTier = urlTier;
        document.querySelectorAll('#sas-tier-tabs button').forEach(btn => {
            btn.classList.toggle('active', btn.getAttribute('data-tier') === urlTier);
        });
    }
    setInterval(refreshPositionsOnly, POS_REFRESH_MS);
    setInterval(scanCycle, SCAN_MS);
    countdownTimer = setInterval(function() { tickCountdown(); tickFullCountdown(); }, 1000);
    tickCountdown();
})();
